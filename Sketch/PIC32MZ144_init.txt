TRISC = 0; 
ANSELC = 0;

ANSELD = 0;

ANSELE = 0;

TRISDbits.TRISD0 = 0; //U2TX

TRISDbits.TRISD11 = 0; //U1TX
TRISCbits.TRISC13 = 0; //U5TX
TRISCbits.TRISC14 = 0; //U3TX

// UART Mapping I/O setting
U1RXR = 0b0010;
U2RXR = 0b0000;
U3RXR = 0b0010;
U4RXR = 0b0000;
U5RXR = 0b0011;

RPD0R = 0b0010; //U2TX
RPD4R = 0b0010; //U4RX
RPD11R = 0b0001; //U1TX
RPC13R = 0b0011; //U5TX

// Initialization of ports
